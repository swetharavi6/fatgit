pipeline {
    agent any
    environment {
        DOCKERHUB_CREDENTIALS = credentials('docker-hub-id') // ID from Jenkins Credentials
        IMAGE_NAME = "your-dockerhub-username/prime-app"
    }
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/swetharavi6/fatgit.git'
            }
        }
        stage('Test') {
            steps {
                bat 'mvn test'
            }
        }
        stage('Build') {
            steps {
                bat 'mvn clean package'
            }
        }
        stage('Docker Build') {
            steps {
                bat "docker build -t ${IMAGE_NAME}:${BUILD_NUMBER} ."
            }
        }
        stage('Push to Docker Hub') {
            steps {bat "echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin"
                bat "docker push ${IMAGE_NAME}:${BUILD_NUMBER}"
            }
        }
    }
}
